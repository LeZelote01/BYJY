# Requirements pour le Chiffreur de Fichiers Personnel
# ============================================

# Cryptographie et Sécurité
cryptography>=41.0.0           # Chiffrement AES-256 via Fernet et PBKDF2

# Modules Python standard utilisés:
# - tkinter : interface graphique (inclus avec Python)
# - tkinter.ttk : widgets modernes
# - tkinter.filedialog : dialogues de sélection de fichiers
# - tkinter.messagebox : boîtes de dialogue
# - tkinter.scrolledtext : zone de texte avec ascenseurs
# - threading : exécution en arrière-plan pour l'UI
# - json : sauvegarde de l'historique des opérations
# - hashlib : fonctions de hachage
# - secrets : génération sécurisée de nombres aléatoires
# - datetime : gestion des timestamps
# - pathlib : manipulation des chemins
# - base64 : encodage des clés cryptographiques
# - zipfile : compression pour le chiffrement de dossiers
# - shutil : opérations sur les fichiers et dossiers
# - os : opérations système

# Version Python recommandée: 3.8+

# Installation:
# pip install -r requirements.txt

# Note sur tkinter:
# Sur certaines distributions Linux, tkinter peut nécessiter une installation séparée:
# Ubuntu/Debian: sudo apt-get install python3-tk
# CentOS/RHEL: sudo yum install tkinter ou sudo dnf install python3-tkinter
# Arch Linux: sudo pacman -S tk

# Pour développement (optionnel):
# pillow>=10.0.0                # Support d'images dans l'interface
